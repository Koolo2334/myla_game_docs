        async def show_chara_list(self, interaction: discord.Interaction):
            """キャラクター一覧の現在のページを表示します。"""
            chara_list_page = self.character_list[self.current_page * 10 : self.current_page * 10 + 10]
            description = "".join([f'**{c[0] + 1}.** {c[1].name} lv.{c[1].lv}\\n' for c in chara_list_page])
            embed = discord.Embed(title="キャラクターを選択", description=description.rstrip("\\n") or "キャラクターがいません", color=discord.Colour.from_rgb(0, 0, 255))
            view = self.ViewSelectCharacter(self, chara_list_page)
            await interaction.response.edit_message(embed=embed, view=view)

        async def update_chara_list(self, interaction: discord.Interaction):
            """フィルターとソートを適用してキャラクター一覧を更新・表示する。"""
            all_charas_data = game.utils.get_player_all_characters_data(self.party_obj.user.id)
            character_list: list[list] = []
            for i, c in enumerate(all_charas_data):
                c_data = game.utils.get_character_data_by_id(c["id"])
                if (not self.fil_rarity or c_data["rarity"] in self.fil_rarity) and \
                   (not self.fil_job or c_data["job"] in self.fil_job) and \
                   (not self.fil_elem or c_data["elem"] in self.fil_elem):
                    character_list.append([i, game.utils.get_player_character(self.party_obj.user.id, c["uuid"])])

            sort_cat_name, sort_cat_id = self.sort_cat.split("-", 1)
            sort_key_map = { "lv": lambda x: x[1].lv, "rarity": lambda x: x[1].rarity, "status": lambda x: x[1].status[int(sort_cat_id)] }
            if sort_cat_name in sort_key_map:
                character_list.sort(key=sort_key_map[sort_cat_name], reverse=self.sort_reverse)

            self.character_list = character_list
            self.current_page = 0
            self.max_page = math.ceil(len(character_list) / 10) - 1
            await self.show_chara_list(interaction)

        async def update_stone_list(self, interaction: discord.Interaction):
            """フィルターを適用してストーン一覧を更新・表示する。"""
            all_stones_data = game.utils.get_player_all_stones_data(self.party_obj.user.id)
            stone_list_lock, stone_list_normal = [], []
            for i, s_data in enumerate(all_stones_data):
                # 部位フィルターと選択されたフィルターを適用
                if (s_data["type"] == self.type_id) and \
                   (not self.fil_type or s_data["type"] in self.fil_type) and \
                   (not self.fil_set or s_data["set"] in self.fil_set):
                    s_obj = game.utils.get_player_stone(self.party_obj.user.id, s_data["uuid"])
                    if s_obj.is_lock:
                        stone_list_lock.append([i, s_obj])
                    else:
                        stone_list_normal.append([i, s_obj])

            stone_list_lock.sort(key=lambda x: x[1].lv, reverse=True)
            stone_list_normal.sort(key=lambda x: x[1].lv, reverse=True)
            self.stone_list = stone_list_lock + stone_list_normal
            
            self.current_page = 0
            self.max_page = math.ceil(len(self.stone_list) / 10) - 1
            await self.show_stone_list(interaction)

    async def add_character(self, interaction: discord.Interaction, character: game.PlayerCharacter):
        """
        キャラクターインベントリから選択されたキャラクターをパーティーに追加/更新します。

    async def show_party_edit(self, interaction: discord.Interaction):
        """現在のパーティー編集状況をEmbedとViewで表示します。"""
        embed = discord.Embed(title="パーティー編成", color=discord.Colour.blue())
        view = self.ViewConfirmEdit(self)

        for i in range(4):
            chara = self.party.characters[i] if i < len(self.party.characters) else None
            if chara:
                # キャラクター情報のフィールドを追加
                value = f"**武器**: {chara.weapon.name if chara.weapon else 'なし'}\\n"
                for j, stone in enumerate(chara.stone):
                     value += f"**ストーン{j+1}**: {stone.name if stone else 'なし'}\\n"
                embed.add_field(name=f"スロット{i+1}: {chara.name} Lv.{chara.lv}", value=value, inline=False)
                # 編集ボタンを追加
                view.add_item(discord.ui.Button(label=f"武器{i+1}", custom_id=f"weapon-{i}"))
                for j in range(4):
                    view.add_item(discord.ui.Button(label=f"ストーン{i+1}-{j+1}", custom_id=f"stone-{i}-{j}"))
            else:
                embed.add_field(name=f"スロット{i+1}: 空き", value="キャラクターを選択してください", inline=False)
            
            view.add_item(discord.ui.Button(label=f"キャラ{i+1}", custom_id=f"chara-{i}"))

        # ボタンのコールバックを動的に設定
        for child in view.children:
            if isinstance(child, discord.ui.Button) and child.custom_id:
                async def button_callback(interaction: discord.Interaction, custom_id=child.custom_id):
                    parts = custom_id.split('-')
                    action_type = parts[0]
                    self.party_slot = int(parts[1])
                    if action_type == "chara":
                        await self.select_character(interaction)
                    elif action_type == "weapon":
                        await self.select_weapon(interaction)
                    elif action_type == "stone":
                        self.stone_slot = int(parts[2])
                        await self.select_stone(interaction, self.stone_slot) # type_idを渡す必要があるが、どのstone typeか？
                child.callback = button_callback
        
        await interaction.response.edit_message(embed=embed, view=view)

    async def update_party_list(self, interaction: discord.Interaction, is_send: bool = True):
        """
        プレイヤーのパーティーデータを読み込み、一覧表示を開始します。
        このメソッドがパーティー編成機能の初期エントリーポイントとなります。
        """
        if is_send:
            self.user = interaction.user

        self.party_list = game.utils.get_player_all_parties(self.user.id)
        
        # 新規作成オプションを追加
        if len(self.party_list) < 10:
            new_party_button = discord.ui.Button(label="新規作成", style=discord.ButtonStyle.success)
            async def new_party_callback(interaction: discord.Interaction):
                self.party = game.Party([None, None, None, None])
                await self.show_party_edit(interaction)
            new_party_button.callback = new_party_callback
        
        if not self.party_list and len(self.party_list) < 10:
             # パーティーが一つもなければ、即座に新規作成画面へ
            self.party = game.Party([None, None, None, None])
            view = self.ViewConfirmEdit(self)
            view.add_item(new_party_button)
            await interaction.response.send_message(embed=discord.Embed(title="パーティーを新規作成"), view=view)
            await self.show_party_edit(interaction)
        else:
            view = self.ViewSelectParty(self, self.party_list)
            if len(self.party_list) < 10:
                view.add_item(new_party_button)
            await self.show_party_list(interaction, is_send)
