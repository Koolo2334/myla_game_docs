    async def end(self):
        """戦闘終了時の処理をオーバーライド。キャラクター育成素材を報酬として付与します。"""

            embed = discord.Embed(title="勝利！", description=description, color=discord.Colour.from_rgb(0, 255, 255))
        else:
            embed = discord.Embed(title="敗北", description="再挑戦しよう", color=discord.Colour.from_rgb(255, 0, 0))

        await self.interaction.edit_original_response(embed=embed, view=None)
        PvECharaDungeon.players_id.remove(self.player.id)
        del self

    async def start_battle(self, interaction: discord.Interaction, values: list[str]):
        """戦闘開始処理をオーバーライド。このダンジョン固有の敵パーティーを生成します。"""
        # プレイヤーのパーティーレベルに合わせて敵のレベルを決定
        party = game.utils.get_player_party(interaction.user.id, values[0])
        lv = max(chara.lv for chara in party.characters)

        # 敵パーティーを生成
        enemies_object = [
            game.EnemyCharacter(20001, lv),
            game.EnemyCharacter(20002, lv),
            game.EnemyCharacter(20001, lv)
        ]
        self.enemy_party = game.EnemyParty(enemies_object)

        # 親クラスの戦闘開始処理を呼び出す
        return await super().start_battle(interaction, values)

    async def end(self):
        """戦闘終了時の処理をオーバーライド。ストーンとストーン強化素材を報酬として付与します。"""
        if self.battle.winner == self.player_party:
            if game.utils.give_stamina(self.player.id, -30) is None:
                description = "スタミナが足りないため、報酬を獲得できません"
                embed = discord.Embed(title="勝利！", description=description, color=discord.Colour.from_rgb(0, 255, 255))
                await self.interaction.edit_original_response(embed=embed, view=None)
            else:
                # ストーン強化素材を付与
                game.utils.give_item(self.player.id, 2, 25)

                # ランダムなストーンを複数個生成・付与
                self.stone_list = []
                for _ in range(random.choices([2,3], [3,1])[0]):
                    stone = game.Stone.create(random.choice(list(game.StoneSet)), random.choice(list(game.StoneType)))
                    game.utils.give_player_stone(self.player.id, stone)
                    self.stone_list.append(stone)
                
                # 獲得したストーンを確認するためのUIを表示
                view = self.ViewSelectStone(self, self.stone_list)
                item_name = game.utils.get_item_data_by_id(2)["name"]
                description = f'{item_name} を **25** 個 獲得した'
                embed = discord.Embed(title="勝利！", description=description, color=discord.Colour.from_rgb(0, 255, 255))
                await self.interaction.edit_original_response(embed=embed, view=view)
        else:
            embed = discord.Embed(title="敗北", description="再挑戦しよう", color=discord.Colour.from_rgb(255, 0, 0))
            await self.interaction.edit_original_response(embed=embed, view=None)
        
        PvEStoneDungeon.players_id.remove(self.player.id)
        del self

class DungeonCmd(commands.Cog):
    """ダンジョン関連のスラッシュコマンドを定義するCog。"""

    def __init__(self, bot: commands.Bot):
        """Cogを初期化し、スタミナ回復タスクを開始します。"""
        self.bot = bot
        self.stamina_charge.start()


    @app_commands.command(name="dungeon_character", description="キャラクター育成ダンジョンに潜る")
    async def dungeon_character(self, int: discord.Interaction):
        """キャラクター育成ダンジョンへの挑戦を開始します。"""
        if game.utils.get_player_stamina(int.user.id) < 20:
            await int.response.send_message("スタミナが足りません", ephemeral=True)
            return
        await PvECharaDungeon().select_party(int)


    @app_commands.command(name="dungeon_weapon", description="武器育成ダンジョンに潜る")
    async def dungeon_weapon(self, int: discord.Interaction):
        """武器育成ダンジョンへの挑戦を開始します。"""
        if game.utils.get_player_stamina(int.user.id) < 20:
            await int.response.send_message("スタミナが足りません", ephemeral=True)
            return
        await PvEWeaponDungeon().select_party(int)


    @app_commands.command(name="dungeon_stone", description="ストーン獲得ダンジョンに潜る")
    async def dungeon_stone(self, int: discord.Interaction):
        """ストーン獲得ダンジョンへの挑戦を開始します。"""
        if game.utils.get_player_stamina(int.user.id) < 30:
            await int.response.send_message("スタミナが足りません", ephemeral=True)
            return
        await PvEStoneDungeon().select_party(int)

    
    @tasks.loop(minutes=6)
    async def stamina_charge(self):
        """6分ごとに全プレイヤーのスタミナを1回復するバックグラウンドタスク。"""
        async with aiofiles.open(utils.sys.get_game_path("data/player.json"), encoding="utf-8") as f:
            c = await f.read()
            player_data = json.loads(c)

        for id in player_data:
            player_data[id]["stamina"] = min(player_data[id]["stamina"] + 1, 180)

        async with aiofiles.open(utils.sys.get_game_path("data/player.json"), "w", encoding="utf-8") as f:
            await f.write(json.dumps(player_data, indent=2))

async def setup(bot: commands.Bot):
    """CogをBotに登録するためのセットアップ関数。"""
    await bot.add_cog(DungeonCmd(bot))
