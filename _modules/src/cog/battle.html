class ViewEnemySpawn(discord.ui.View):
    """敵が出現した際に表示されるViewコンポーネント。
    
    「戦う」ボタンを提供し、押下されると戦闘準備フローを開始します。
    タイムアウトは1800秒（30分）で、時間内に誰も戦闘を開始しなければ消滅します。
    """
        
    def __init__(self, pve_obj: PvE, *, timeout: float | None = 180):
        """ViewEnemySpawnのコンストラクタ。

        Args:
            pve_obj (PvE): このViewに対応するPvEセッションのインスタンス。
            timeout (float | None, optional): Viewがタイムアウトするまでの秒数。デフォルトは180秒。
        """
        super().__init__(timeout=timeout)
        self.pve_obj = pve_obj

class PvE:
    """PvE（Player versus Environment）戦闘のフロー全体を管理するクラス。
    
    パーティー選択UIの表示、戦闘の開始、戦闘画面の更新、終了までを一貫して扱います。
    このクラス自体は汎用的なフローを定義し、具体的な敵の構成や報酬は
    `dungeon.py`などのサブクラスで実装されます。
    """

    # 現在、何らかの戦闘フローに参加しているプレイヤーのIDリスト
    # プレイヤーが複数の戦闘を同時に開始することを防ぐ
    players_id: list[int] = []


    def __init__(self):
        """PvEセッションを初期化します。"""
        self.player: discord.User | discord.Member | None = None
        self.player_party: game.PlayerParty | None = None
        self.enemy_party: game.EnemyParty | None = None
        self.battle: game.Battle | None = None
        self.interaction: discord.Interaction | None = None

            async def callback(select: discord.ui.Select, interaction: discord.Interaction):
                """パーティーが選択された際のコールバック関数。
                
                選択したユーザーが戦闘開始者本人であることを確認し、戦闘を開始します。
                """
                # 操作者が戦闘開始者でなければ無視
                if interaction.user.id != select.view.pve_obj.player.id:
                    await interaction.response.defer()
                    return
                
                # 戦闘開始処理を呼び出す
                await select.view.pve_obj.start_battle(interaction, select.values)




        

        def __init__(self, pve_obj, party_list: list[game.Party], *, timeout: float | None = 180):
            """ViewSelectPartyのコンストラクタ。

            Args:
                pve_obj (PvE): このViewが紐づくPvEセッションのインスタンス。
                party_list (list[game.Party]): 選択肢となるプレイヤーのパーティーリスト。
                timeout (float | None, optional): Viewがタイムアウトするまでの秒数。
            """
            super().__init__(timeout=timeout)

            self.pve_obj = pve_obj

            select = self.SelectParty()

            # プレイヤーが所持するパーティーを選択肢として追加
            for party in party_list:
                select.add_option(label=f'{party.characters[0].name} 他{len(party.characters) - 1}人', value=party.uuid)

            self.add_item(select)

    async def end(self):
        """戦闘終了時の処理。
        
        このメソッドは基底クラスでは具体的な実装を持たず、サブクラスでオーバーライドされることを
        想定しています。報酬の計算や表示などがサブクラスで行われます。
        """
        # 戦闘が勝利で終わったか判定
        if self.battle.winner == self.player_party:
            embed = discord.Embed(title="勝利！", description="おめでとう！", color=discord.Colour.from_rgb(0, 255, 255))
        else:
            embed = discord.Embed(title="敗北", description="再挑戦しよう", color=discord.Colour.from_rgb(255, 0, 0))
        
        # メッセージを編集して結果を表示
        await self.interaction.edit_original_response(embed=embed, view=None)

        # プレイヤーを戦闘中リストから削除
        PvE.players_id.remove(self.player.id)
        del self

    async def select_party(self, interaction: discord.Interaction):
        """プレイヤーに戦闘で使用するパーティーを選択させるUIを表示します。

        Args:
            interaction (discord.Interaction): ユーザーとのインタラクション。
        """
        # プレイヤーが既に他の戦闘に参加している場合はエラーメッセージを返す
        if interaction.user.id in PvE.players_id:
            await interaction.response.send_message("あなたはすでに他の戦闘に参加しています", ephemeral=True)
            return
        
        # プレイヤー情報をセッションに保存し、戦闘中リストに追加
        self.player = interaction.user
        PvE.players_id.append(self.player.id)
        self.interaction = interaction
        
        # プレイヤーのパーティーデータを読み込む
        party_list = game.utils.get_player_parties(self.player.id)
        
        # パーティーが一つもなければエラーメッセージを返す
        if not party_list:
            await interaction.response.send_message("パーティーがありません", ephemeral=True)
            return

        # パーティー選択用のViewとEmbedを作成して送信
        view = self.ViewSelectParty(self, party_list)
        embed = discord.Embed(title="パーティーを選択してください", description="戦闘に使用するパーティーを選んでください", color=discord.Colour.from_rgb(0,0,255))
        await interaction.response.send_message(embed=embed, view=view, ephemeral=True)

    async def start_battle(self, interaction: discord.Interaction, values: list[str]):
        """選択されたパーティーで戦闘を開始し、最初の戦闘画面を表示します。

        Args:
            interaction (discord.Interaction): ユーザーとのインタラクション。
            values (list[str]): 選択されたパーティーのUUIDを含むリスト。
        """
        # 選択されたパーティーを取得
        self.player_party = game.utils.get_player_party(self.player.id, values[0])

        # Battleインスタンスを生成して戦闘を開始
        self.battle = game.Battle(self.player_party, self.enemy_party)
        
        # 元のメッセージ（「敵がスポーンした！」）を更新して戦闘画面を表示
        await self.show_battle_screen(interaction)

    async def show_battle_screen(self, interaction: discord.Interaction):
        """現在の戦闘状況をEmbedとして表示します。

        Args:
            interaction (discord.Interaction): ユーザーとのインタラクション。
        """
        # 戦闘が終了しているかチェック
        if self.battle.is_settled:
            await self.end()
            return
        
        # 戦闘画面のEmbedを作成
        embed = discord.Embed(title="戦闘中！", description="", color=discord.Colour.from_rgb(255,255,0))

        # 味方パーティの情報をフィールドに追加
        player_party_str = ""
        for i, chara in enumerate(self.player_party.characters):
            player_party_str += f'**[{i + 1}] {chara.name}** {math.ceil(chara.hp)}/{math.ceil(chara.get_effective_status(game.Status.HP))}\\n'
        embed.add_field(name="[味方パーティー]", value=player_party_str.rstrip("\\n"), inline=False)

        # 敵パーティの情報をフィールドに追加
        enemy_party_str = ""
        for i, chara in enumerate(self.enemy_party.characters):
            enemy_party_str += f'**[{i + 1}] {chara.name}** {math.ceil(chara.hp)}/{math.ceil(chara.get_effective_status(game.Status.HP))}\\n'
        embed.add_field(name="[敵パーティー]", value=enemy_party_str.rstrip("\\n"), inline=False)
        
        # 行動ログをフッターに表示
        if hasattr(self, "log") and self.log:
             embed.set_footer(text="[ログ]\n" + "\n".join(self.log))
        
        # 行動順と現在の行動キャラクターを表示
        order_str = ""
        active_chara_name = self.battle.get_active_character().name
        for chara in self.battle.order:
            if chara.name == active_chara_name:
                order_str += f'**>> {chara.name}**\\n'
            else:
                order_str += f'{chara.name}\\n'
        embed.add_field(name="[行動順]", value=order_str.rstrip("\\n"))

        # 戦闘操作用のViewを作成
        view = self.ViewBattle(self, self.enemy_party)

        # ephemeralでない場合は元のメッセージを編集、ephemeralの場合は新しいメッセージを送信
        if interaction.message.flags.ephemeral:
            await interaction.response.edit_message(embed=embed, view=view)
        else:
            await self.interaction.edit_original_response(embed=embed, view=view)

    async def act(self, interaction: discord.Interaction, values: list[str]):
        """プレイヤーの行動を処理し、戦闘を1ターン進めます。

        Args:
            interaction (discord.Interaction): ユーザーとのインタラクション。
            values (list[str]): 選択されたターゲットのUUIDを含むリスト。
        """
        # ターゲットUUIDからターゲットキャラクターオブジェクトを取得
        target = [c for c in self.enemy_party.characters if c.uuid == values[0]][0]

        # battleオブジェクトに行動を処理させる
        action_data = self.battle.act(target)

        # 行動ログを生成
        self.log = []
        log_str = f'{action_data.actor.name} は {action_data.target.name} に {action_data.action_type.name} をした！'
        if action_data.value is not None:
            log_str += f' **{math.ceil(action_data.value)}** のダメージ！' if action_data.action_type == game.ActionType.ATTACK else f' **{math.ceil(action_data.value)}** 回復！'
        if action_data.is_crit:
            log_str += ' **CRITICAL!!**'
        self.log.append(log_str)
        
        # スキルが発動した場合、そのログも追加
        for skill in action_data.list_skill:
             skill_log = f'  >> Skill[{skill.skill_type.name_ja}]が発動！'
             self.log.append(skill_log)

        # 次の戦闘画面を表示
        await self.show_battle_screen(interaction)

class BattleCmd(commands.Cog):
    """戦闘関連のコマンドとバックグラウンドタスクを定義するCog。"""

    def __init__(self, bot: commands.Bot):
        """Cogを初期化し、敵を定期的にスポーンさせるタスクを開始します。

        Args:
            bot (commands.Bot): Botのインスタンス。
        """
        self.bot = bot
        self.spawn_enemy.start()


    @tasks.loop(minutes=30)
    async def spawn_enemy(self):
        """30分ごとに実行されるタスク。
        
        設定されたワールドチャンネルにランダムな敵パーティーを生成し、
        `ViewEnemySpawn`と共にメッセージを送信します。
        """
        # ワールドチャンネルIDを取得
        world_id = 992425654992339014
        channel = self.bot.get_channel(world_id)

        if channel:
            # 現在チャンネルにいるメンバーのレベル帯を取得
            lv_list = [game.utils.get_player_level(member.id) for member in channel.members if not member.bot]
            if not lv_list:
                lv_list = [1]
            
            min_lv = max(min(lv_list) - 5, 1)
            max_lv = max(lv_list)
            
            # 敵データを読み込み、レベル帯に応じた敵を生成
            with open(utils.sys.get_game_path("asset/data/character.json"), "r", encoding="utf-8") as f:
                character_data = json.load(f)

            min_count, max_count = 1, 3
            # プレイヤーの最大レベルに応じて敵の数を調整
            if max_lv >= 51: min_count, max_count = 4, 5
            elif max_lv >= 41: max_count = 5
            elif max_lv >= 21: min_count = 3
            elif max_lv >= 11: min_count = 2
            elif max_lv >= 6: max_count = 4

            enemies_info = random.choices(character_data["enemy"], weights=[i["weight"] for i in character_data["enemy"]], k=random.randint(min_count, max_count))
            
            enemies_object = [game.EnemyCharacter(e["id"], random.randint(min_lv, max_lv)) for e in enemies_info]
            enemy_party = game.EnemyParty(enemies_object)

            # EmbedとViewを作成してメッセージを送信
            embed_description = "\\n".join([f'**{e.name}** lv.{e.lv}' for e in enemy_party.characters])
            pve_obj = PvE()
            pve_obj.enemy_party = enemy_party
            view = ViewEnemySpawn(pve_obj, timeout=1800)
            embed = discord.Embed(title="モンスターがスポーンした！", description=embed_description, color=discord.Colour.from_rgb(0,255,0))
            await channel.send(embed=embed, view=view, silent=True)
