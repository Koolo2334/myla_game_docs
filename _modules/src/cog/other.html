class OtherCmd(commands.Cog):
    """補助的なコマンドとタスクをまとめたCogクラス。"""

    def __init__(self, bot: commands.Bot):
        """Cogを初期化し、バックグラウンドタスクを開始します。

        Args:
            bot (commands.Bot): Botのインスタンス。
        """
        self.bot = bot
        self.reset_gold_hour_count()
        self.loop_reset_gold_hour_count.start()

    
    @app_commands.command(name="daily", description="デイリー報酬を獲得する")
    async def daily(self, int: discord.Interaction):
        """デイリーログインボーナス（ゴールド）を受け取ります。1日1回まで。

        Args:
            int (discord.Interaction): インタラクションオブジェクト。
        """
        if game.utils.daily_login(int.user.id):
            await int.response.send_message(f'**60G** 獲得しました', ephemeral=True)
        else:
            await int.response.send_message(f'すでに獲得済みです', ephemeral=True)


    @app_commands.command(name="gold", description="ゴールドの所持数を確認する")
    async def gold(self, int: discord.Interaction):
        """現在のゴールド所持数を表示します。

        Args:
            int (discord.Interaction): インタラクションオブジェクト。
        """
        count = game.utils.get_player_gold_count(int.user.id)
        await int.response.send_message(f'**{count} G**', ephemeral=True)

    async def get_item_count(self, interaction: discord.Interaction, select: discord.ui.Select):

    async def start(self, int: discord.Interaction):
        """ゲーム開始時に、初期キャラクター2体を付与します。1アカウントにつき1回のみ実行可能です。

        Args:
            int (discord.Interaction): インタラクションオブジェクト。
        """
        if game.utils.has_player_character_by_id(int.user.id, 10001):
            await int.response.send_message("すでに所有しています", ephemeral=True)
            return
        
        game.utils.give_player_character(int.user.id, 10001, 1)
        game.utils.give_player_character(int.user.id, 10003, 1)

        await int.response.send_message(f'初期キャラクターを獲得しました', ephemeral=True)

    def reset_gold_hour_count(self):
        """全プレイヤーの「1時間あたりのゴールド獲得量」をリセットします。"""
        with open(utils.sys.get_game_path("data/gold.json"), "r+", encoding="utf-8") as f:
            gold_data = json.load(f)
            for id in gold_data:
                gold_data[id]["in_hour"] = 0
            f.seek(0)
            json.dump(gold_data, f, indent=2)
            f.truncate()




    @tasks.loop(time=[datetime.time(hour=i) for i in range(24)])
    async def loop_reset_gold_hour_count(self):
        """毎時0分に`reset_gold_hour_count`を実行するバックグラウンドタスク。"""
        self.reset_gold_hour_count()
