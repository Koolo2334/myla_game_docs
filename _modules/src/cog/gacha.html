    async def show_result(self, interaction: discord.Interaction):
        """ガチャで排出された全アイテムの一覧をEmbedで表示します。"""
        description = ""
        rarity_map = {0: "【⭐️⭐️⭐️】", 1: "【⭐️⭐️⭐️⭐️】", 2: "【⭐️⭐️⭐️⭐️⭐️】"}
        type_map = {"c": "キャラ", "w": "武器"}
        
        for i, item in enumerate(self.items):
            item_data = game.utils.get_character_data_by_id(item["id"]) if item["type"] == "c" else game.utils.get_weapon_data_by_id(item["id"])
            description += f'{rarity_map.get(self.rarities[i], "")}**{item_data["name"]}** [{type_map[item["type"]]}][{self.get_types[i]}]\\n'
        
        embed = discord.Embed(title="結果", description=description, color=discord.Colour.from_rgb(0, 255, 255))
        await interaction.response.edit_message(embed=embed, attachments=[], view=None)

        self.current_page = (self.current_page - 1 + self.max_page + 1) % (self.max_page + 1)

class GachaCmd(commands.Cog):
    """ガチャ関連のスラッシュコマンドを定義するCog。"""
    def __init__(self, bot: commands.Bot):
        self.bot = bot

    @app_commands.command(name="gacha", description="ガチャを引く")
    async def gacha(self, int: discord.Interaction):
        """ガチャ画面を開始します。"""
        await Gacha().show_gacha(int)

    """CogをBotに登録するためのセットアップ関数。"""
