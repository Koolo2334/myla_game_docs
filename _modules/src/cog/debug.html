    @app_commands.command(name="get_character_by_id", description="[デバッグ用]指定したIDのキャラクターを獲得する")
    @app_commands.describe(id="キャラクターID")
    async def get_character_by_id(self, int: discord.Interaction, id: int):
        """指定したIDのキャラクターをコマンド実行者に付与します。管理者のみ実行可能。

        Args:
            int (discord.Interaction): インタラクションオブジェクト。
            id (int): 付与するキャラクターのID。
        """
        # 管理者IDチェック
        if int.user.id != 123456789012345678:
            await int.response.send_message("権限がありません", ephemeral=True)
            return
        
        # 敵キャラクターIDでないかチェック
        if id >= 20000:
            await int.response.send_message("無効なIDです", ephemeral=True)
            return
        
        # すでに所持しているかチェック
        if game.utils.has_player_character_by_id(int.user.id, id):
            await int.response.send_message("すでに所有しています", ephemeral=True)
            return
        
        character = game.utils.get_character_data_by_id(id)

        # キャラクターIDが存在するかチェック
        if character is None:
            await int.response.send_message("無効なIDです", ephemeral=True)
            return
        
        # キャラクターを付与
        game.utils.give_player_character(int.user.id, id, 1)

        await int.response.send_message(f'**{character["name"]}**を獲得しました', ephemeral=True)

    @app_commands.command(name="view_stone", description="[デバッグ用]ストーンを見る")
    @app_commands.describe(set_id="セットID", type_id="タイプID")
    async def view_stone(self, int: discord.Interaction, set_id: int, type_id: int):
        """指定したセットIDとタイプIDでストーンを仮想的に生成し、その情報を表示します。
           管理者のみ実行可能。

        Args:
            int (discord.Interaction): インタラクションオブジェクト。
            set_id (int): ストーンのセットID (-1でランダム)。
            type_id (int): ストーンのタイプID (-1でランダム)。
        """
        if int.user.id != 123456789012345678:
            await int.response.send_message("権限がありません", ephemeral=True)
            return

        # IDのバリデーションとランダム選択
        try:
            stone_set = random.choice(list(game.StoneSet)) if set_id == -1 else game.StoneSet.get_by_id(set_id)
            stone_type = random.choice(list(game.StoneType)) if type_id == -1 else game.StoneType.get_by_id(type_id)
        except IndexError:
            await int.response.send_message("無効なIDです", ephemeral=True)
            return

        # ストーンを生成してEmbedで表示
        stone = game.Stone.create(stone_set, stone_type)
        file = discord.File(fp=utils.sys.get_game_path(f"asset/img/{stone.img_path}"), filename="stone.png")
        embed = discord.Embed(title=stone.name, description=f"3セット：{stone.set.status.name_ja} +15%", color=discord.Colour.from_rgb(255, 255, 0))
        embed.set_image(url="attachment://stone.png")
        value = "".join([f"{sop.option.name_ja}　　{sop.get_display_value()}\\n" for sop in stone.sops])
        embed.add_field(name=f"{stone.mop.option.name_ja}　　{stone.mop.get_display_value()}", value=value.rstrip("\\n"))
        await int.response.send_message(embed=embed, file=file)
