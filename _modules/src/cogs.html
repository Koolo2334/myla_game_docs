# -*- coding: utf-8 -*-
"""
Discord Botの拡張機能（Cog）をロードするためのエントリーポイントモジュール。
Bot起動時にこのファイルが読み込まれ、`INITIAL_EXTENSIONS`に定義された
すべてのCogがBotに登録されます。
"""
import discord
from discord import app_commands
from discord.interactions import Interaction
from discord.ext import commands, tasks

# Cogが格納されているPythonパッケージへのパス
COG_DIR = 'game.src.cog.'

# Bot起動時にロードするCogのリスト
# 新しいコマンドファイルを追加した場合は、ここにそのファイル名を追加する
INITIAL_EXTENSIONS = [
    'battle',
    'debug',
    'gacha',
    'inventory',
    'dungeon',
    'party',
    'other'
]

async def setup(bot: commands.Bot):
    """
    Botのセットアップ時に自動的に呼び出される関数。
    INITIAL_EXTENSIONSにリストアップされたすべてのCogを非同期でロードします。

    Args:
        bot (commands.Bot): discord.pyのBotインスタンス。
    """
    #リスト内の各Cogを順にロードする
    for cog in INITIAL_EXTENSIONS:
        await bot.load_extension(COG_DIR + cog)
