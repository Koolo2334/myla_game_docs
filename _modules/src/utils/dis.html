class DefaultSelect(discord.ui.Select):
    """
    コールバック関数と引数をコンストラクタで受け取れるようにしたカスタムセレクトメニュー。
    """
    def __init__(self,
                 callback_func: typing.Awaitable,
                 callback_args: tuple,
                 *args, **kwargs):
        super().__init__(*args, **kwargs)
        self.callback_func = callback_func
        self.callback_args = callback_args

        await self.callback_func(interaction, self.values, *self.callback_args)
