# -*- coding: utf-8 -*-
"""
戦闘中の継続的なステータス変化（バフ・デバフ）を表現するための
データクラスを定義するモジュール。
"""
import dataclasses
from . import status, effect_type




[ドキュメント]
@dataclasses.dataclass
class Effect:
    """
    キャラクターに付与される継続的な効果（バフ・デバフ）を表すデータクラス。
    """
    # 対象となるステータス
    status: "status.Status"
    # 効果のタイプ（加算か乗算か）
    type: "effect_type.EffectType"
    # 効果の値
    value: float | int
    # 効果が持続するターン数 (-1は永続)
    duration: int = 1
    # スキルなどで同じ効果を管理・削除するための識別タグ
    tag: str = "Default"
