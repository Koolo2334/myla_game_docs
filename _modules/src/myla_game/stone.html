class StoneOption(enum.Enum):
    """
    ストーンが持ちうるオプション（ステータス効果）の種類を定義するEnum。
    
    各メンバーは `asset/data/stone.json` 内の "option" リストに対応しています。
    Enumの初期化時に、対応するIDのデータを読み込み、プロパティとして保持します。
    """

    HP_ADD = 0
    HP_RATE = 1
    ATK_ADD = 2
    ATK_RATE = 3
    DEF_ADD = 4
    DEF_RATE = 5
    CRIT_RATE = 6
    CRIT_DMG = 7
    SPD = 8

    def __init__(self, id):
        """
        Enumメンバーを初期化し、JSONから詳細データを読み込みます。

        Args:
            id (int): `stone.json` 内のオプションID。
        """
        super().__init__()

        with open(utils_bot.sys.get_game_path("asset/data/stone.json"), "r", encoding="utf-8") as f:
            stone_data = json.load(f)

        stone_option_data = stone_data["option"]
        option_data = [s for s in stone_option_data if s["id"] == id]
        option = option_data[0] if option_data else None

        self.id = option["id"]
        self.name_ja = option["name"]
        self.sfx = option["sfx"]  # 表示用の接尾辞 (例: '%')
        self.dp = option["dp"]    # 表示時の小数点以下の桁数
        self.main_base = option["main_base"] # メインオプションの初期値
        self.main_incr = option["main_incr"] # メインオプションの強化ごとの上昇値
        self.sub_incr = option["sub_incr"]   # サブオプションの強化値の範囲
        self.status = status.Status.get_by_id(option["status"]) # 対応するステータス
        self.effect_type = option["effect_type"] # 効果の種類 ('add' or 'mul')

class OptionData:
    """メインオプションとサブオプションの基底クラス。"""

    def __init__(self, option: StoneOption, value: int|float):
        """
        オプションデータを初期化します。

        Args:
            option (StoneOption): オプションの種類。
            value (int | float): オプションの現在値。
        """
        self.option = option
        self.value = value

    


[ドキュメント]
    def get_display_value(self) -> str:
        """
        表示用にフォーマットされたオプションの値を文字列として返します。
        例: "15.3%" や "125"。

        Returns:
            str: フォーマット済みの値。
        """
        value = self.value
        if self.option.sfx == "%":
            value *= 100

        # Decimalを使用して浮動小数点数演算の誤差を避け、正確な丸め処理を行う
        buf = Decimal(str(math.floor(value * 10 ** self.option.dp) / 10 ** self.option.dp))
        buf = buf.quantize(Decimal(1)) if buf == buf.to_integral() else buf.normalize()
        
        return f'{buf}{self.option.sfx}'

class MainOptionData(OptionData):
    """ストーンのメインオプションを表すクラス。"""

    def __init__(self, option: StoneOption, value: int|float|None = None):
        """
        メインオプションを初期化します。値が指定されない場合は、基本初期値を使用します。

        Args:
            option (StoneOption): オプションの種類。
            value (int | float | None): オプションの現在値。新規作成時はNone。
        """
        if value is None:
            super().__init__(option, option.main_base)
        else:
            super().__init__(option, value)
    


[ドキュメント]
    def enhance(self):
        """オプションを1段階強化します。固定の上昇値が加算されます。"""
        dp = self.option.dp
        if self.option.sfx == "%":
            dp += 2
        self.value = round(self.value + self.option.main_incr, dp)

class SubOptionData(OptionData):
    """ストーンのサブオプションを表すクラス。"""

    def __init__(self, option: StoneOption, value: int|float|None = None):
        """
        サブオプションを初期化します。値が指定されない場合は、ランダムな初期値が設定されます。

        Args:
            option (StoneOption): オプションの種類。
            value (int | float | None): オプションの現在値。新規作成時はNone。
        """
        if value is None:
            super().__init__(option, random.choice(option.sub_incr))
        else:
            super().__init__(option, value)
    


[ドキュメント]
    def enhance(self):
        """オプションを1段階強化します。ランダムな上昇値が加算されます。"""
        dp = self.option.dp
        if self.option.sfx == "%":
            dp += 2
        self.value = round(self.value + random.choice(self.option.sub_incr), dp)

class Stone:
    """
    ストーン本体を表すクラス。
    生成、強化、経験値管理などのロジックを持ちます。
    """

    def __init__(self, lv: int, uuid: str, set: StoneSet, type: StoneType, mop: MainOptionData, sops: list[SubOptionData], exp: int = 0, is_lock: bool = False):
        """
        Stoneオブジェクトを初期化します。

        Args:
            lv (int): 現在のレベル。
            uuid (str): ストーンの一意なID。
            set (StoneSet): セット効果の種類。
            type (StoneType): 部位の種類。
            mop (MainOptionData): メインオプション。
            sops (list[SubOptionData]): サブオプションのリスト。
            exp (int): 現在のレベルでの蓄積経験値。
            is_lock (bool): ロック状態（粉砕・売却防止）。
        """
        self.lv = lv
        self.uuid = uuid
        self.set = set
        self.type = type
        self.mop = mop
        self.sops = sops
        self.exp = exp
        self.is_lock = is_lock

        self.name = set.name_pfx + self.type.name_ja
        self.img_path = "stone/" + set.img_pfx + self.type.img_id + ".png"

    @classmethod
    def create(cls, set: StoneSet, type: StoneType) -> "Stone":
        """
        新しいストーンをランダムに生成するファクトリメソッド。

        Args:
            set (StoneSet): 生成するストーンのセット。
            type (StoneType): 生成するストーンの部位。

        Returns:
            Stone: 新しく生成されたStoneオブジェクト。
        """
        # メインオプションを確率に基づいて決定
        mop_id = random.choices(type.mop_ids, type.mop_weights)[0]

        # サブオプションを確率に基づいて決定（メインオプションとは重複しない）
        sops = []
        op_qty = random.choices([3,4], [3,2])[0] # サブオプションの初期個数を決定 (3個 or 4個)
        sop_weights = type.sop_weights.copy()
        available_sop_ids = type.sop_ids.copy()

        if mop_id in available_sop_ids:
            idx = available_sop_ids.index(mop_id)
            sop_weights.pop(idx)
            available_sop_ids.pop(idx)

        for _ in range(op_qty):
            chosen_sop_id = random.choices(available_sop_ids, sop_weights)[0]
            sops.append(SubOptionData(StoneOption.get_by_id(chosen_sop_id)))
            
            idx = available_sop_ids.index(chosen_sop_id)
            sop_weights.pop(idx)
            available_sop_ids.pop(idx)

        mop = MainOptionData(StoneOption.get_by_id(mop_id))

        return cls(0, str(uuid.uuid4()), set, type, mop, sops)


    
    def _get_stone_lvup_exp_req(self, lv: int) -> int:
        """指定レベルから次のレベルに上がるために必要な経験値を取得します。"""
        with open(utils_bot.sys.get_game_path("asset/data/stone.json"), "r", encoding="utf-8") as f:
            stone_data = json.load(f)
        if not (0 <= lv < 15): return -1
        return stone_data["lvup_exp_table"][lv]
    
    def _get_stone_lvup_exp_sum(self, lv: int) -> int:
        """レベル1から指定レベルまで上げるために必要な累計経験値を取得します。"""
        with open(utils_bot.sys.get_game_path("asset/data/stone.json"), "r", encoding="utf-8") as f:
            stone_data = json.load(f)
        if not (0 <= lv <= 15): return -1
        return sum(stone_data["lvup_exp_table"][:lv])

    def lvup(self) -> bool:
        """
        ストーンを1レベルアップさせ、オプションを強化します。
        3の倍数レベルに達した際、サブオプションが追加または強化されます。

        Returns:
            bool: レベルアップに成功したかどうか。
        """
        if self.lv >= 15: return False
        
        self.lv += 1
        self.mop.enhance()

        if self.lv % 3 == 0:
            if len(self.sops) < 4:
                # 新しいサブオプションを追加
                current_op_ids = {self.mop.option.id} | {sop.option.id for sop in self.sops}
                available_sops = [(sop_id, weight) for sop_id, weight in zip(self.type.sop_ids, self.type.sop_weights) if sop_id not in current_op_ids]
                if available_sops:
                    new_op_id = random.choices([s[0] for s in available_sops], [s[1] for s in available_sops])[0]
                    self.sops.append(SubOptionData(StoneOption.get_by_id(new_op_id)))
            else:
                # 既存のサブオプションをランダムに強化
                random.choice(self.sops).enhance()
        return True

    def calc_lvup(self, exp_item_value: int) -> tuple[int, int]:
        """
        指定された数の経験値アイテムを使用した場合のレベルアップ結果を計算（プレビュー）します。
        実際の経験値やレベルは変更しません。

        Args:
            exp_item_value (int): 使用する経験値アイテムの数。

        Returns:
            tuple[int, int]: (上昇するレベル数, 実際に消費されるアイテム数)。
        """
        lvup_count = 0
        exp_item_count = 0
        exp = self.exp
        current_lv = self.lv

        # まず、現在の余剰経験値でレベルアップできるかチェック
        while (req_exp := self._get_stone_lvup_exp_req(current_lv)) != -1 and exp >= req_exp:
            exp -= req_exp
            current_lv += 1
            lvup_count += 1
        
        # 次に、アイテムを1つずつ消費して計算
        while current_lv < 15 and exp_item_count < exp_item_value:
            exp += 250
            exp_item_count += 1
            while (req_exp := self._get_stone_lvup_exp_req(current_lv)) != -1 and exp >= req_exp:
                exp -= req_exp
                current_lv += 1
                lvup_count += 1
        
        return lvup_count, exp_item_count

    def give_exp(self, exp_amount: int) -> int:
        """
        ストーンに経験値を付与し、必要であればレベルアップ処理を呼び出します。

        Args:
            exp_amount (int): 付与する経験値量。

        Returns:
            int: 上昇したレベル数。
        """
        self.exp += exp_amount
        lvup_count = 0
        while (req_exp := self._get_stone_lvup_exp_req(self.lv)) != -1 and self.exp >= req_exp:
            self.exp -= req_exp
            self.lvup()
            lvup_count += 1
        
        if self.lv >= 15:
            self.exp = 0
            
        return lvup_count

    def convert_to_exp(self) -> int:
        """
        このストーンを粉砕した場合に得られる総経験値を計算します。
        (基本経験値 + レベルアップに必要な累計経験値 + 現在の余剰経験値)

        Returns:
            int: 総経験値。
        """
        return 1500 + self._get_stone_lvup_exp_sum(self.lv) + self.exp
