import enum

class Element(enum.Enum):
    """
    キャラクターの属性を表す列挙型。
    各属性は、回復、バフ、デバフに対する基本効果量を持ちます。
    """
    GRASS = (0, "草", 0.15, 0.10, 0.05)
    ELECTRIC = (1, "雷", 0.05, 0.10, 0.15)
    ROCK = (2, "岩", 0.10, 0.15, 0.05)
    WATER = (3, "水", 0.15, 0.05, 0.10)
    FIRE = (4, "火", 0.05, 0.15, 0.10)
    AIR = (5, "風", 0.10, 0.05, 0.15)


    def __init__(self, id, name_ja, heal, buff, debuff):
        """
        Args:
            id (int): 属性ID。
            name_ja (str): 日本語名。
            heal (float): 回復効果倍率。
            buff (float): バフ効果倍率。
            debuff (float): デバフ効果倍率。
        """
        self.id = id
        self.name_ja = name_ja
        self.heal = heal
        self.buff = buff
        self.debuff = debuff

    @classmethod
    def get_by_id(cls, id: int) -> "Element":
        """IDから対応するElementメンバーを返します。"""
        return [i for i in cls if i.id == id][0]
