class Status(enum.Enum):
    """
    キャラクターや装備が持つパラメータを表す列挙型。
    各ステータスは、ID、日本語名、レベルアップで成長するか、武器に付くかなどの属性を持ちます。
    """
    HP = (0, "体力", True, True, 1)
    ATK = (1, "攻撃", True, True, 1)
    DEF = (2, "防御", True, True, 1)
    C_RATE = (3, "会心率", False, False, 0.0)
    C_DMG = (4, "会心ダメージ", False, False, 0.0)
    SPD = (5, "速度", False, False, 1)


    def __init__(self, id, name_ja, is_increasable, is_weapon, min_value):
        """
        Args:
            id (int): ステータスID。
            name_ja (str): 日本語名。
            is_increasable (bool): レベルアップで成長するステータスか。
            is_weapon (bool): 武器の基本ステータスとして存在するか。
            min_value (float): ステータスの最小値。
        """
        self.id = id
        self.name_ja = name_ja
        self.is_increasable = is_increasable
        self.is_weapon = is_weapon
        self.min_value = min_value

    def get_display_value(self, value: float) -> str:
        """値を表示用の文字列（整数またはパーセンテージ）にフォーマットします。"""
        if self.id in [0, 1, 2, 5]:
            return str(int(value))
        else:
            return f'{round(value * 100, 1)}%'

    def calc_value(self, value: float) -> float:
        """値を計算用の型（整数または浮動小数点数）に変換します。"""
        if self.id in [0, 1, 2, 5]:
            return int(value)
        else:
            return round(value, 3)

    @classmethod
    def get_by_id(cls, id: int) -> "Status":
        """IDから対応するStatusメンバーを返します。"""
        return [i for i in cls if i.id == id][0]
