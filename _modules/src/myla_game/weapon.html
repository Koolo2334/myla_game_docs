class Weapon:
    """
    キャラクターが装備する武器を表すクラス。
    """
    def __init__(self, id: int, lv: int, uuid: str, exp: int = 0):
        """
        Weaponインスタンスを初期化します。
        IDに基づきasset/data/weapon.jsonから基本データを読み込み、レベルに応じたステータスを計算します。

        Args:
            id (int): 武器のユニークID。
            lv (int): 武器のレベル。
            uuid (str): インスタンス固有のUUID。
            exp (int): 現在の累積経験値。
        """
        self.id = id
        self.lv = lv
        self.uuid = uuid
        self.exp = exp

        # JSONから武器の基本データを取得
        weapon_data = utils.get_weapon_data_by_id(id)

        self.name = weapon_data["name"]
        self.status = weapon_data["status"]
        self.rarity = weapon_data["rarity"]
        # レベルとレアリティに基づいてステータスを計算
        for i, s_val in enumerate(self.status):
            stat = status.Status.get_by_id(i)
            if stat.is_increasable:
                self.status[i] = stat.calc_value(s_val * (1 + (0.083 + self.rarity * 0.001) * (self.lv - 1)))
            else:
                self.status[i] = stat.calc_value(s_val)

        self.job = job.Job.get_by_id(weapon_data["job"]) # 装備可能な職業
        self.skill = skill.Skill.get_by_id(weapon_data["skill"]) # 武器スキル
        self.img_path = weapon_data["img_path"]
