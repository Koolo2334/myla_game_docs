# -*- coding: utf-8 -*-
"""
戦闘中の単一のアクション（攻撃、回復など）に関する情報を格納するための
データクラスを定義するモジュール。
"""
import dataclasses
import typing
# 型ヒントのためにモジュールをインポートするが、循環参照を避けるため文字列として扱う
from . import battle, character, action_type, skill_data, effect



[ドキュメント]
@dataclasses.dataclass()
class ActionData:
    """
    戦闘中の1回のアクションに関するすべての情報を保持するデータクラス。
    このオブジェクトは、あるキャラクターが行動を起こしてから、その結果が確定し、
    関連するスキルが発動するまで、情報伝達の器として機能します。
    """
    # 現在の戦闘インスタンス
    battle: typing.Optional["battle.Battle"] = None
    # 行動の主体となるキャラクター
    actor: typing.Optional["character.Character"] = None
    # 行動の対象となるキャラクター
    target: typing.Optional["character.Character"] = None
    # 行動の種類 (攻撃、回復など)
    action_type: typing.Optional["action_type.ActionType"] = None
    # 行動によって与えられたダメージや回復量
    value: typing.Optional[int] = None
    # 行動によって付与されたエフェクト（バフ・デバフ）
    effect: typing.Optional["effect.Effect"] = None
    # この行動がクリティカルヒットだったか
    is_crit: typing.Optional[bool] = None
    # この行動によってターゲットを倒したか
    is_defeat: typing.Optional[bool] = None
    # この行動をトリガーとして発動したスキルのリスト
    list_skill: typing.Optional[list["skill_data.SkillData"]] = None
    # この行動によって戦闘が終了したか
    is_settled: bool = False
