`cog`層や他の`myla_game`層のモジュールは、原則としてこのモジュールを通じて
データにアクセスすることで、データ永続化のロジックを抽象化します。

def get_character_data_by_id(id: int) -> dict | None:
    """
    指定されたIDを持つキャラクターの静的データを `asset/data/character.json` から取得します。

    Args:
        id (int): 取得したいキャラクターのID。

    Returns:
        dict | None: 見つかったキャラクターのデータ辞書。見つからない場合はNone。
    """
    with open(utils.sys.get_game_path("asset/data/character.json"), "r", encoding="utf-8") as f:
        character_data = json.load(f)
    
    # プレイヤーキャラクターと敵キャラクターの両方を検索対象に含める
    characters = character_data["player"] + character_data["enemy"]
    characters = [c["data"] for c in characters if c["id"] == id]
    
    return characters[0] if characters else None

def has_player_character_by_id(user_id: int, id: int) -> bool:
    """
    プレイヤーが指定されたIDのキャラクターを所持しているかを確認します。

    Args:
        user_id (int): プレイヤーのDiscordユーザーID。
        id (int): 確認したいキャラクターのID。

    Returns:
        bool: 所持している場合はTrue、そうでない場合はFalse。
    """
    with open(utils.sys.get_game_path("data/character.json"), "r", encoding="utf-8") as f:
        character_data = json.load(f)
    
    characters = character_data.get(str(user_id), [])
    return any(c["id"] == id for c in characters)

def get_player_character(user_id: int, uuid: str, weapon_uuid: str | None = None):
    """
    プレイヤーが所持する特定のキャラクターのインスタンスを生成して返します。
    武器UUIDが指定された場合は、その武器も装備した状態でインスタンス化します。

    Args:
        user_id (int): プレイヤーのDiscordユーザーID。
        uuid (str): 取得したいキャラクターのユニークID。
        weapon_uuid (str | None, optional): 装備させる武器のユニークID。Defaults to None.

    Returns:
        character.PlayerCharacter | None: 生成されたキャラクターオブジェクト。見つからない場合はNone。
    """
    with open(utils.sys.get_game_path("data/character.json"), "r", encoding="utf-8") as f:
        character_data = json.load(f)
    
    characters = character_data.get(str(user_id), [])
    target_char_data = [c for c in characters if c["uuid"] == uuid]
    
    if not target_char_data: return None
    
    char_data = target_char_data[0]
    
    wep = None
    if weapon_uuid:
        wep = get_player_weapon(user_id, weapon_uuid)
        # 武器の職業とキャラクターの職業が一致しない場合は装備させない
        if wep and wep.job != character.PlayerCharacter(user_id, **char_data).job:
            return None

    # ストーンを装備した状態でインスタンス化
    stones = [get_player_stone(user_id, s_uuid) for s_uuid in char_data.get("stone", []) if s_uuid]
    
    return character.PlayerCharacter(user_id, weapon=wep, stone=stones, **char_data)

def get_player_stone(user_id: int, uuid: str):
    """
    プレイヤーが所持する特定のストーンのインスタンスを生成して返します。

    Args:
        user_id (int): プレイヤーのDiscordユーザーID。
        uuid (str): 取得したいストーンのユニークID。

    Returns:
        stone.Stone | None: 生成されたストーンオブジェクト。見つからない場合はNone。
    """
    with open(utils.sys.get_game_path("data/stone.json"), "r", encoding="utf-8") as f:
        stone_data = json.load(f)
        
    stones = stone_data.get(str(user_id), [])
    target_stone_data = [s for s in stones if s["uuid"] == uuid]
    
    if not target_stone_data: return None
    
    s_data = target_stone_data[0]
    # JSONから読み込んだデータを元に各オブジェクトを再構築
    mop = stone.MainOptionData(stone.StoneOption.get_by_id(s_data["mop"]["option"]), s_data["mop"]["value"])
    sops = [stone.SubOptionData(stone.StoneOption.get_by_id(sop["option"]), sop["value"]) for sop in s_data["sops"]]
    
    return stone.Stone(
        set=stone.StoneSet.get_by_id(s_data["set"]),
        type=stone.StoneType.get_by_id(s_data["type"]),
        mop=mop, sops=sops, **{k:v for k,v in s_data.items() if k not in ["set", "type", "mop", "sops"]}
    )

def get_player_party(user_id: int, uuid: str):
    """
    プレイヤーが編成した特定のパーティーのインスタンスを生成して返します。
    パーティー内のキャラクターや装備も再帰的に読み込み、完全にインスタンス化します。

    Args:
        user_id (int): プレイヤーのDiscordユーザーID。
        uuid (str): 取得したいパーティーのユニークID。

    Returns:
        party.PlayerParty | None: 生成されたパーティーオブジェクト。見つからない場合はNone。
    """
    with open(utils.sys.get_game_path("data/party.json"), "r", encoding="utf-8") as f:
        party_data = json.load(f)
        
    parties = party_data.get(str(user_id), [])
    target_party_data = [p for p in parties if p["uuid"] == uuid]
    
    if not target_party_data: return None
    
    p_data = target_party_data[0]
    characters_in_party = [get_player_character(user_id, c["uuid"], c.get("weapon")) for c in p_data["characters"] if c]
    
    return party.PlayerParty(user_id, characters_in_party, p_data["uuid"])

def give_gold(user_id: int, value: int) -> bool:
    """
    プレイヤーの所持ゴールドを増減させます。

    Args:
        user_id (int): プレイヤーのDiscordユーザーID。
        value (int): 増減させるゴールドの量。負の値も可。

    Returns:
        bool: 操作が成功した場合はTrue。ゴールドがマイナスになるなど失敗した場合はFalse。
    """
    with open(utils.sys.get_game_path("data/gold.json"), "r", encoding="utf-8") as f:
        gold_data = json.load(f)
    
    player_gold = gold_data.get(str(user_id), {"value": 0, "in_hour": 0})
    
    if player_gold["value"] + value < 0:
        return False
        
    player_gold["value"] += value
    gold_data[str(user_id)] = player_gold
    
    with open(utils.sys.get_game_path("data/gold.json"), "w", encoding="utf-8") as f:
        json.dump(gold_data, f, indent=2)
    return True

def give_player_character(user_id: int, id: int, value: int) -> int:
    """
    プレイヤーにキャラクターを付与します。
    既に所持している場合は、凸数を進めます。

    Args:
        user_id (int): プレイヤーのDiscordユーザーID。
        id (int): 付与するキャラクターのID。
        value (int): 付与する数（通常は1）。

    Returns:
        int: 操作後のキャラクターの凸数 (新規獲得の場合は0)。
             -1の場合はエラーまたは最大凸数を超えていることを示す。
    """
    with open(utils.sys.get_game_path("data/character.json"), "r", encoding="utf-8") as f:
        character_data = json.load(f)
        
    player_chars = character_data.get(str(user_id), [])
    
    existing_char = [c for c in player_chars if c["id"] == id]
    
    if existing_char:
        # 既に所持している場合 -> 凸数を進める
        char_to_update = existing_char[0]
        if char_to_update["count"] >= 7: return -1 # 最大凸
        char_to_update["count"] += value
        count = char_to_update["count"]
    else:
        # 新規獲得の場合
        new_char = {
            "id": id,
            "lv": 1,
            "exp": 0,
            "uuid": str(uuid.uuid4()),
            "count": 0,
            "stone": [None, None, None, None]
        }
        player_chars.append(new_char)
        count = 0
        
    character_data[str(user_id)] = player_chars
    
    with open(utils.sys.get_game_path("data/character.json"), "w", encoding="utf-8") as f:
        json.dump(character_data, f, indent=2)
        
    return count

def give_player_stone(user_id: int, stone_obj: stone.Stone):
    """
    プレイヤーにストーンを追加または更新します。
    同じUUIDのストーンが既に存在する場合は上書きされ、レベルやロック状態の更新に使用できます。

    Args:
        user_id (int): プレイヤーのDiscordユーザーID。
        stone_obj (stone.Stone): 保存するStoneオブジェクト。
    """
    with open(utils.sys.get_game_path("data/stone.json"), "r", encoding="utf-8") as f:
        stone_data = json.load(f)
    
    player_stones = stone_data.get(str(user_id), [])
    
    # オブジェクトをJSONシリアライズ可能な辞書に変換
    stone_dict = {
        "uuid": stone_obj.uuid, "lv": stone_obj.lv, "exp": stone_obj.exp, "is_lock": stone_obj.is_lock,
        "set": stone_obj.set.id, "type": stone_obj.type.id,
        "mop": {"option": stone_obj.mop.option.id, "value": stone_obj.mop.value},
        "sops": [{"option": sop.option.id, "value": sop.value} for sop in stone_obj.sops]
    }
    
    # 既存のストーンを探して更新、なければ追加
    found = False
    for i, s in enumerate(player_stones):
        if s["uuid"] == stone_obj.uuid:
            player_stones[i] = stone_dict
            found = True
            break
            
    if not found:
        player_stones.append(stone_dict)
        
    stone_data[str(user_id)] = player_stones
    
    with open(utils.sys.get_game_path("data/stone.json"), "w", encoding="utf-8") as f:
        json.dump(stone_data, f, indent=2)

def give_player_party(user_id: int, party_obj: party.Party):
    """
    プレイヤーのパーティー編成を保存または更新します。

    Args:
        user_id (int): プレイヤーのDiscordユーザーID。
        party_obj (party.Party): 保存するPartyオブジェクト。
    """
    with open(utils.sys.get_game_path("data/party.json"), "r", encoding="utf-8") as f:
        party_data = json.load(f)
        
    player_parties = party_data.get(str(user_id), [])
    
    # PartyオブジェクトをJSONシリアライズ可能な辞書に変換
    party_dict = {
        "uuid": party_obj.uuid,
        "characters": [
            {"uuid": c.uuid, "weapon": c.weapon.uuid if c.weapon else None} if c else None
            for c in party_obj.characters
        ]
    }

    found = False
    for i, p in enumerate(player_parties):
        if p["uuid"] == party_obj.uuid:
            player_parties[i] = party_dict
            found = True
            break
    
    if not found:
        player_parties.append(party_dict)
        
    party_data[str(user_id)] = player_parties
    
    with open(utils.sys.get_game_path("data/party.json"), "w", encoding="utf-8") as f:
        json.dump(party_data, f, indent=2)
