class Character:
    """
    すべてのキャラクター（プレイヤー、敵）の基本となる基底クラス。
    ステータス、装備、戦闘中の状態などを管理します。
    """
    def __init__(self, id: int, lv: int, uuid: str, weapon: "weapon.Weapon" | None = None, stone: list["stone.Stone"] | None = None):
        """
        キャラクターを初期化します。
        IDに基づきasset/data/character.jsonから基本データを読み込み、レベルに応じたステータス計算を行います。

        Args:
            id (int): キャラクターのユニークID。
            lv (int): キャラクターのレベル。
            uuid (str): インスタンス固有のUUID。
            weapon (weapon.Weapon | None): 装備している武器。
            stone (list[stone.Stone] | None): 装備しているストーンのリスト。
        """
        self.id = id
        self.lv = lv
        self.uuid = uuid

        # JSONからキャラクターの基本データを取得
        character_data = utils.get_character_data_by_id(id)

        self.name = character_data["name"]
        self.status = character_data["status"]
        self.rarity = character_data["rarity"]
        # レベルとレアリティに基づいてステータスを計算
        for i, s_val in enumerate(self.status):
            stat = status.Status.get_by_id(i)
            if stat.is_increasable:
                self.status[i] = stat.calc_value(s_val * (1 + (0.083 + self.rarity * 0.005) * (self.lv - 1)))
            else:
                self.status[i] = stat.calc_value(s_val)
        
        self.job = job.Job.get_by_id(character_data["job"])
        self.elem = element.Element.get_by_id(character_data["elem"])
        self.heal_rate = self.elem.heal
        self.buff_rate = self.elem.buff
        self.debuff_rate = self.elem.debuff
        self.img_path = character_data["img_path"]

        self.weapon = weapon
        self.stone = stone or [None, None, None, None]
        
        # 戦闘中に設定されるプロパティ
        self.battle: "battle.Battle" | None = None
        self.party: "party.Party" | None = None
        self.party_index: int | None = None
        self.opponent: "party.Party" | None = None
        
        # 行動順を決定するためのカウント
        self.count = 0
        self.last_count = 0

        # 現在のHP
        self.hp = self.get_effective_status(status.Status.HP) # HPは初期状態で最大値
        # 付与されているエフェクト（バフ・デバフ）
        self.effects: list["effect.Effect"] = []

    def get_effective_status(self, status_enum: "status.Status") -> float:
        """
        指定されたステータスの最終的な実効値を計算して返します。
        キャラクターの基本ステータスに、武器、ストーン、継続効果（バフ/デバフ）による
        すべての補正を適用した値を算出します。

        Args:
            status_enum (status.Status): 計算対象のステータス。

        Returns:
            float: 計算後のステータス値。
        """
        value = self.status[status_enum.id]

        # 武器による補正
        if self.weapon:
            value += self.weapon.get_status(status_enum)

        # ストーンによる補正
        stone_add, stone_mul, set_count = 0, 1.0, 0
        active_sets = {}
        for s in self.stone:
            if s:
                active_sets[s.set] = active_sets.get(s.set, 0) + 1
                if s.mop.option.status == status_enum:
                    if s.mop.option.effect_type == "add": stone_add += s.mop.value
                    else: stone_mul += s.mop.value
                for sop in s.sops:
                    if sop.option.status == status_enum:
                        if sop.option.effect_type == "add": stone_add += sop.value
                        else: stone_mul += sop.value
        # セット効果の適用
        for s_set, count in active_sets.items():
            if count >= 3 and s_set.status == status_enum:
                stone_mul += 0.15

        value = value * stone_mul + stone_add

        # バフ・デバフによる補正
        effect_add, effect_rate = 0, 1.0
        for effect_instance in self.effects:
            if effect_instance.status == status_enum:
                if effect_instance.type == effect_type.EffectType.ADD:
                    effect_add += effect_instance.value
                elif effect_instance.type == effect_type.EffectType.RATE:
                    effect_rate += effect_instance.value
        
        value = value * effect_rate + effect_add

        return max(status_enum.min_value, status_enum.calc_value(value))

    def attack(self, action_data_obj: "action_data.ActionData", ref_status: "status.Status", rate: float = 1.0, should_crit: bool = True, should_trigger_skill: bool = True):
        """
        ターゲットに攻撃を行います。クリティカル判定、ダメージ計算、ダメージ適用、
        そして攻撃をトリガーとするスキルの発動までを処理します。

        Args:
            action_data_obj (action_data.ActionData): この攻撃に関する情報を格納するオブジェクト。
            ref_status (status.Status): ダメージ計算の基礎となるステータス（通常はATK）。
            rate (float): ダメージ倍率。
            should_crit (bool): クリティカルが発生するか。
            should_trigger_skill (bool): この攻撃でスキルがトリガーされるか。
        """
        if not self.is_alive(): return

        # クリティカル判定
        is_crit = should_crit and random.random() < self.get_effective_status(status.Status.C_RATE)
        action_data_obj.is_crit = is_crit

        # ダメージ計算
        dmg = self.get_effective_status(ref_status)
        if is_crit:
            dmg *= (1 + self.get_effective_status(status.Status.C_DMG))
        dmg = math.floor(dmg * rate)

        # ターゲットにダメージを適用
        action_data_obj.target.damaged(action_data_obj, dmg)
        if action_data_obj.is_settled: return

        # スキルトリガー判定
        if should_trigger_skill and self.weapon:
            if trigger_type.TriggerType.ATTACK in self.weapon.skill.trigger_type:
                self.weapon.skill.func(action_data_obj)
            if is_crit and trigger_type.TriggerType.CRIT in self.weapon.skill.trigger_type:
                self.weapon.skill.func(action_data_obj)
            if action_data_obj.is_defeat and trigger_type.TriggerType.DEFEAT in self.weapon.skill.trigger_type:
                self.weapon.skill.func(action_data_obj)

    def act(self, target: Character) -> "action_data.ActionData":
        """
        キャラクターの1ターン分の行動を実行します。
        職業に応じて攻撃または回復を行います。

        Args:
            target (Character): 行動の対象。

        Returns:
            action_data.ActionData: 実行された行動の結果。
        """
        ad = action_data.ActionData(battle=self.battle, actor=self, target=target, list_skill=[])
        
        # 職業に応じて基本行動を決定
        if self.job.action_type == action_type.ActionType.ATTACK:
            ad.action_type = action_type.ActionType.ATTACK
            self.attack(ad, status.Status.ATK)
        elif self.job.action_type == action_type.ActionType.HEAL:
            ad.action_type = action_type.ActionType.HEAL
            self.heal(ad, status.Status.HP)

        if self.battle.is_settled: return ad

        # エフェクトの持続時間を更新
        self.effects = [e for e in self.effects if e.duration - 1 != 0]
        for e in self.effects: e.duration -= 1
        
        return ad
