class Battle:
    """
    2つのパーティー間の戦闘を管理・シミュレートするクラス。
    ターン制の戦闘ロジック、行動順の決定、勝敗判定などを担当します。
    """

    def __init__(self, team1: "party.Party", team2: "party.Party"):
        """
        Battleインスタンスを初期化し、戦闘を開始します。

        Args:
            team1 (party.Party): チーム1のパーティーオブジェクト。
            team2 (party.Party): チーム2のパーティーオブジェクト。
        """
        # 各キャラクターに戦闘インスタンスと敵パーティー情報を設定
        for c in team1.characters:
            c.set_battle(self)
            c.set_opponent(team2)
        for c in team2.characters:
            c.set_battle(self)
            c.set_opponent(team1)

        self.parties = [team1, team2]

        # 全キャラクターを速度に基づいてソートし、初期行動順を決定
        # countは行動済みかどうかを示す値で、未行動のキャラが優先される
        all_characters = self.parties[0].characters + self.parties[1].characters
        self.order = sorted(all_characters, key=lambda c: c.count + 10000 / c.get_effective_status(status.Status.SPD))
        
        # 最初の行動キャラクターを決定し、そのキャラクターの行動カウントを進める
        self.active_character = self.order[0]
        self.active_character.add_count()
        
        # 再度ソートして次の行動順を確定
        self.order.sort(key=lambda c: c.count + 10000 / c.get_effective_status(status.Status.SPD))
        
        self.is_settled = False # 戦闘が終了したか
        self.winner = None      # 勝者パーティー

    def get_active_character(self) -> "character.Character":
        """
        現在行動するべきキャラクターを返します。

        Returns:
            character.Character: 現在アクティブなキャラクター。
        """
        return self.order[0]

    def calc_order(self):
        """
        現在の全キャラクターの行動カウントと速度に基づいて、行動順リストを再計算・ソートします。
        """
        self.order.sort(key=lambda c: c.count + math.floor(10000 / c.get_effective_status(status.Status.SPD)))

    def check_settled(self) -> bool:
        """
        戦闘の勝敗が決定したかを確認します。
        どちらかのパーティーの全キャラクターが戦闘不能になった場合に終了と判定します。

        Returns:
            bool: 戦闘が終了していればTrue、そうでなければFalse。
        """
        for p in self.parties:
            # パーティー内の全キャラクターが生存していないかチェック
            if all(not c.is_alive() for c in p.characters):
                # 敵対パーティーを勝者として設定
                self.winner = [i for i in self.parties if i != p][0]
                self.is_settled = True
                return True
        
        self.is_settled = False
        return False

    def act(self, target: "character.Character") -> "action_data.ActionData":
        """
        現在アクティブなキャラクターの行動を処理し、戦闘を1ターン進めます。

        Args:
            target (character.Character): 行動の対象となるキャラクター。

        Returns:
            action_data.ActionData: 実行された行動とその結果をまとめたデータ。
        """
        # アクティブキャラクターに行動を実行させる
        action_data = self.active_character.act(target)

        # 戦闘不能になったキャラクターを行動順リストから除外
        self.order = [c for c in self.order if c.is_alive()]

        # 勝敗判定
        if not self.check_settled():
            # 戦闘が続く場合、次の行動キャラクターを決定する
            self.calc_order()
            self.active_character = self.order[0]
            self.active_character.add_count()
            self.calc_order()

        return action_data
