"""
ゲーム内に登場するすべてのスキル（武器スキル、ジョブスキルなど）を定義するモジュール。

`Skill` Enumは、各スキルのID、名前、トリガー条件、そして具体的な効果を実装した関数を
一元管理します。これにより、スキルデータとロジックが密結合し、管理が容易になります。
"""

class Skill(enum.Enum):
    """
    ゲーム内のすべてのスキルを定義するEnum。

    各メンバーは以下の情報を持つタプルとして定義されます:
    - id (int): スキルの一意なID。
    - name_ja (str): スキル名（日本語）。
    - trigger_type (list[TriggerType]): スキルが発動する条件のリスト。
    - func (function): スキルの効果を実装した関数。

    各スキルロジック関数は、`action_data.ActionData`オブジェクトを引数に取り、
    戦闘状態の変更や追加効果の発生などを行います。

    def func_shaman_1(ad: "action_data.ActionData"):
        """
        【シャーマン１：自己再生】のスキルロジック。
        トリガー：回復行動時 (HEAL)

        術者のHPを、術者の最大HPの20%分回復させます。
        このスキルによる回復では、さらなるスキルはトリガーされません。

        Args:
            ad (ActionData): 元となった行動のデータ。
        """
        # スキル発動による新たな行動データを作成
        ad_skill = action_data.ActionData(battle=ad.battle, actor=ad.actor, target=ad.actor, action_type=action_type.ActionType.HEAL, list_skill=[])

        # 回復処理を実行
        ad_skill.actor.heal(ad_skill, status.Status.HP, rate=0.2, should_trigger_skill=False)

        # 元の行動データに、このスキルが発動したことを記録
        ad.list_skill.append(skill_data.SkillData(Skill.SHAMAN_1, ad_skill))

    def func_shaman_2(ad: "action_data.ActionData"):
        """
        【シャーマン２】のスキルロジック。
        トリガー：回復行動時 (HEAL)

        回復対象の攻撃力を、術者のバフ効果率に応じて上昇させます。

        Args:
            ad (ActionData): 元となった行動のデータ。
        """
        ad_skill = action_data.ActionData(battle=ad.battle, actor=ad.actor, target=ad.target, action_type=action_type.ActionType.BUFF, list_skill=[])

        # 攻撃力上昇のEffectオブジェクトを作成
        efct = effect.Effect(status.Status.ATK, effect_type.EffectType.RATE, ad.actor.buff_rate)

        # 対象にEffectを付与
        ad_skill.target.add_effect(efct)

        ad_skill.effect = efct
        ad.list_skill.append(skill_data.SkillData(Skill.SHAMAN_2, ad_skill))

    def func_fighter_2(ad: "action_data.ActionData"):
        """
        【ファイター２】のスキルロジック。
        トリガー：攻撃時 (ATTACK)

        主対象以外の敵単体に、攻撃力の20%の追撃を行います。
        この追撃ではスキルはトリガーされません。

        Args:
            ad (ActionData): 元となった行動のデータ。
        """
        # 主対象以外の敵キャラクターをリストアップ
        targets = [i for i in ad.target.party.characters if i.uuid != ad.target.uuid]
        if not targets: return # 他にターゲットがいない場合は終了

        target = random.choice(targets)

        ad_skill = action_data.ActionData(battle=ad.battle, actor=ad.actor, target=target, action_type=action_type.ActionType.ATTACK, list_skill=[])

        # 追撃処理を実行
        ad.actor.attack(ad_skill, status.Status.ATK, rate=0.2, should_trigger_skill=False)

        ad.list_skill.append(skill_data.SkillData(Skill.FIGHTER_2, ad_skill))

    def func_guardian_1(ad: "action_data.ActionData"):
        """
        【ガーディアン１】のスキルロジック。
        トリガー：攻撃時 (ATTACK)

        味方の中で最もHPが低いキャラクターの防御力を、術者の防御力とバフ効果率に応じて2ターンの間上昇させます。

        Args:
            ad (ActionData): 元となった行動のデータ。
        """
        # 最もHPが低い味方をターゲットに設定
        target = min(ad.actor.party.characters, key=operator.attrgetter("hp"))

        ad_skill = action_data.ActionData(battle=ad.battle, actor=ad.actor, target=target, action_type=action_type.ActionType.BUFF, list_skill=[])

        # 防御力上昇のEffectオブジェクトを作成（加算効果）
        efct = effect.Effect(status.Status.DEF, effect_type.EffectType.ADD, math.ceil(ad.actor.get_effective_status(status.Status.DEF) * ad.actor.buff_rate), duration=2)

        ad_skill.target.add_effect(efct)
        ad_skill.effect = efct
        ad.list_skill.append(skill_data.SkillData(Skill.GUARDIAN_1, ad_skill))

    def func_guardian_2(ad: "action_data.ActionData"):
        """
        【ガーディアン２】のスキルロジック。
        トリガー：クリティカル攻撃時 (CRIT)

        術者以外の味方2体に、防御力上昇バフを2ターンの間付与します。

        Args:
            ad (ActionData): 元となった行動のデータ。
        """
        # 術者以外の味方をリストアップ
        targets = [i for i in ad.actor.party.characters if i.uuid != ad.actor.uuid]
        if not targets: return

        # ランダムに2体選ぶ（2体未満の場合は全員）
        targets = random.sample(targets, k=min(len(targets), 2))

        for target in targets:
            efct = effect.Effect(status.Status.DEF, effect_type.EffectType.ADD, math.ceil(ad.actor.get_effective_status(status.Status.DEF) * ad.actor.buff_rate), duration=2)
            target.add_effect(efct)
            ad_skill = action_data.ActionData(battle=ad.battle, actor=ad.actor, target=target, action_type=action_type.ActionType.BUFF, effect=efct, list_skill=[])
            ad.list_skill.append(skill_data.SkillData(Skill.GUARDIAN_2, ad_skill))

    def func_cleric_2(ad: "action_data.ActionData"):
        """
        【クレリック２】のスキルロジック。
        トリガー：クリティカル攻撃時 (CRIT)

        次に行動する味方の攻撃力を、術者のバフ効果率に応じて上昇させます。

        Args:
            ad (ActionData): 元となった行動のデータ。
        """
        # 戦闘の行動順リストから、次の味方を特定
        target = [i for i in ad.battle.order if i.party.uuid == ad.actor.party.uuid][0]

        ad_skill = action_data.ActionData(battle=ad.battle, actor=ad.actor, target=target, action_type=action_type.ActionType.BUFF, list_skill=[])

        efct = effect.Effect(status.Status.ATK, effect_type.EffectType.RATE, ad.actor.buff_rate)
        ad_skill.target.add_effect(efct)
        ad_skill.effect = efct
        ad.list_skill.append(skill_data.SkillData(Skill.CLERIC_2, ad_skill))

    def func_berserker_1(ad: "action_data.ActionData"):
        """
        【バーサーカー１】のスキルロジック。
        トリガー：クリティカル攻撃時 (CRIT)

        攻撃対象の速度を、術者のデバフ効果率に応じて2ターンの間減少させます。

        Args:
            ad (ActionData): 元となった行動のデータ。
        """
        ad_skill = action_data.ActionData(battle=ad.battle, actor=ad.actor, target=ad.target, action_type=action_type.ActionType.DEBUFF, list_skill=[])

        efct = effect.Effect(status.Status.SPD, -effect_type.EffectType.RATE, ad.actor.debuff_rate, duration=2)
        ad_skill.target.add_effect(efct)
        ad.battle.calc_order() # 速度が変化したので行動順を再計算

        ad_skill.effect = efct
        ad.list_skill.append(skill_data.SkillData(Skill.BERSERKER_1, ad_skill))

    def func_berserker_2(ad: "action_data.ActionData"):
        """
        【バーサーカー２】のスキルロジック。
        トリガー：被ダメージ時 (DAMAGED), 回復時 (HEALED)

        自身のHPが30%以下の場合、攻撃力を30%上昇させる永続バフを付与します。
        HPが30%を超えると、このバフは解除されます。

        Args:
            ad (ActionData): 元となった行動のデータ。
        """
        if ad.actor.hp / ad.actor.get_effective_status(status.Status.HP) <= 0.3:
            ad_skill = action_data.ActionData(battle=ad.battle, actor=ad.actor, target=ad.actor, action_type=action_type.ActionType.BUFF, list_skill=[])

            # 解除用のタグを付けた永続バフ(-1)を付与
            efct = effect.Effect(status.Status.HP, effect_type.EffectType.RATE, 0.3, duration=-1, tag="Skill_Berserker_2")
            ad_skill.target.add_effect(efct)
            ad_skill.effect = efct
            ad.list_skill.append(skill_data.SkillData(Skill.BERSERKER_2, ad_skill))
        else:
            # 条件を満たさなくなったらタグで指定してバフを解除
            ad.actor.remove_effect("Skill_Berserker_2")

    def func_assassin_1(ad: "action_data.ActionData"):
        """
        【アサシン１】のスキルロジック。
        トリガー：敵を倒した時 (DEFEAT)

        自身の行動ゲージをリセットし、即座に再行動します。

        Args:
            ad (ActionData): 元となった行動のデータ。
        """
        ad_skill = action_data.ActionData(battle=ad.battle, actor=ad.actor, target=ad.actor, action_type=action_type.ActionType.OTHER, list_skill=[])
        
        # 自身の行動カウントを直前の行動分だけ巻き戻す
        ad_skill.target.count -= ad_skill.target.last_count

        ad.list_skill.append(skill_data.SkillData(Skill.ASSASSIN_1, ad_skill))
